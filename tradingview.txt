//
// 1) 텔레그램 알림: “정규장(RTH) 종가 확정(=16:00 ET)” 순간에만 즉시 1회 발송.
//    - 일봉이 애프터 포함으로 “진짜로 확정(isconfirmed)”되는 시각(애프터 종료)까지 기다리지 않음.
//    - 동시에, 다음날 개장(23:30 KST) 오발송(전일 True sticky) 가능성 차단.
//
// 2) 트레이딩뷰 신호 라벨: 위 (1)과 같은 “정규장 종가 확정” 순간에 즉시 발생.
//    - 애프터장 종료 후에 뜨는 지연 현상 제거.
//
// [오발송(23:30) 근본 원인]
// - request.security(5m, session.islastbar & confirmed)는 “정규장 마지막 5분봉” 이후 다음날 첫 확정 5분봉 전까지
//   마지막 값(True)이 유지(sticky)될 수 있음.
// - 이 상태에서 일봉이 새로 시작(newBarSlot)되면 update가 개장 시점에 True가 되어 알림이 나갈 수 있음.
//
// [해결 방식: 5분봉 “종가 time_close”를 키로 1일 1회 펄스로 변환]
// - rthCloseRaw: 정규장 마지막 5분봉이 확정되는 순간 True
// - rthCloseTs : 그 마지막 5분봉의 time_close(=정규장 종가 시각)
// - lastRthCloseTs와 비교해 “같은 종가를 두 번 처리”할 수 없도록 차단 → 개장 오발송 방지
// ============================================================

sampleStdev(src, len) =>
    m = ta.sma(src, len)
    v = ta.sma((src - m) * (src - m), len)
    math.sqrt(v * len / math.max(len - 1, 1))

rsiWilder(src, len) =>
    ch = ta.change(src)
    up = math.max(ch, 0.0)
    dn = math.max(-ch, 0.0)
    au = ta.rma(up, len)
    ad = ta.rma(dn, len)
    (au == 0 and ad == 0) ? 50.0 :
     (ad == 0) ? 100.0 :
     (au == 0) ? 0.0 :
     100.0 - (100.0 / (1.0 + (au / ad)))

// codeWeight: 기존 코드(0/1/2/3/4/5) + permille 코드(>=100)
codeWeight(code) =>
    switch code
        2 => 1.00
        3 => 0.90
        4 => 0.80
        1 => 0.10
        5 => 0.05
        0 => 0.00
        => (code >= 100 ? math.min(math.max(code / 1000.0, 0.0), 1.0) : 0.00)

codePctStr(code) =>
    if code == 2
        "100"
    else if code == 3
        "90"
    else if code == 4
        "80"
    else if code == 1
        "10"
    else if code == 5
        "5"
    else if code == 0
        "0"
    else if code >= 100
        float p = code / 10.0
        float pRound = math.round(p)
        bool isInt = math.abs(p - pRound) < 1e-6
        isInt ? str.tostring(pRound, "0") : str.tostring(p, "0.0")
    else
        "0"

weightToCode(w) =>
    float wC = math.min(math.max(w, 0.0), 1.0)
    float eps = 0.0005
    if wC <= eps
        0
    else if math.abs(wC - 0.05) < eps
        5
    else if math.abs(wC - 0.10) < eps
        1
    else if math.abs(wC - 0.80) < eps
        4
    else if math.abs(wC - 0.90) < eps
        3
    else if math.abs(wC - 1.00) < eps
        2
    else
        int(math.round(wC * 1000.0))

isHighExposure(code) =>
    codeWeight(code) >= 0.80 - 1e-9

assetNameLabel(code) =>
    float w = codeWeight(code)
    w <= 0 ? "현금" : ("TQQQ(" + codePctStr(code) + "%)")

assetNameAlert(code) =>
    float w = codeWeight(code)
    w <= 0 ? "현금(SGOV, BOXX)" : ("TQQQ(" + codePctStr(code) + "%)")

posIcon(code) =>
    float w = codeWeight(code)
    w <= 0 ? "⚫" : (w <= 0.051 ? "🟡" : (w < 0.80 ? "🟠" : "🟢"))

posColor(code) =>
    float w = codeWeight(code)
    w <= 0 ? color.gray : (w <= 0.051 ? color.yellow : (w < 0.80 ? color.orange : color.green))

fmtPct(x) => (x > 0 ? "+" : "") + str.tostring(x, "0.00") + "%"

stageReason(stage) =>
    switch stage
        1 => "리스크 관리(200일 이격도 분할매도 1차)"
        2 => "리스크 관리(200일 이격도 분할매도 2차)"

        4 => "리스크 관리(200일 이격도 과열)"
        => ""

dirEmoji(curr, prev) =>
    (na(curr) or na(prev) or prev == 0) ? "🟢" : (curr >= prev ? "🟢" : "🔴")

makeAdjustedBy(sym, adj) =>
    parts = str.split(sym, ":")
    n = array.size(parts)
    ex = n >= 2 ? array.get(parts, 0) : "NASDAQ"
    tk = n >= 2 ? array.get(parts, 1) : sym
    ex := str.length(ex) == 0 ? "NASDAQ" : ex
    tk := str.length(tk) == 0 ? sym : tk
    ticker.new(ex, tk, session.regular, adj)

makeAdjusted(sym) =>
    adj = adjustedMode == "splits" ? adjustment.splits : adjustment.dividends
    makeAdjustedBy(sym, adj)

makeRegular(sym) =>
    makeAdjustedBy(sym, adjustment.none)
